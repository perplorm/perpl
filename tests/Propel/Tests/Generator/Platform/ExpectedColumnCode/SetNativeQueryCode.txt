
    /**
     * Filter the query on the enumerated_column column
     *
     * @param mixed $enumeratedColumn The value to use as filter
     * @param string|null $comparison Operator to use for the column comparison, defaults to Criteria::EQUAL
     *
     * @return $this
     */
    public function filterByEnumeratedColumn($enumeratedColumn = null, ?string $comparison = null)
    {
        $resolvedColumn = $this->resolveLocalColumnByName('enumerated_column');
        $binaryOperator = match($comparison) {
            null,
            Criteria::CONTAINS_ALL => Criteria::BINARY_ALL,
            Criteria::CONTAINS_SOME,
            Criteria::IN => Criteria::BINARY_AND,
            Criteria::CONTAINS_NONE => Criteria::BINARY_NONE,
            default => null,
        };

        if ($binaryOperator) {
            if (!$enumeratedColumn && in_array($binaryOperator, [Criteria::BINARY_AND, Criteria::BINARY_ALL], true)) {
                return $this;
            }
            if (!$enumeratedColumn && $binaryOperator === Criteria::BINARY_NONE) {
                $itemBitmask = -1; // none but empty set (Propel-specific behavior)
            } else {
                $valueSet = TableTableMap::getValueSet(TableTableMap::COL_ENUMERATED_COLUMN);
                $itemBitmask = SetColumnConverter::convertToBitmask($enumeratedColumn, $valueSet);
            }

            $this->addUsingOperator($resolvedColumn, $itemBitmask, $binaryOperator);
            if ($binaryOperator === Criteria::BINARY_NONE) {
                $this->addOr($resolvedColumn, null, Criteria::ISNULL);
            }

            return $this;
        }

        $this->addUsingOperator($resolvedColumn, $enumeratedColumn, $comparison);

        return $this;
    }
