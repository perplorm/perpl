
    /**
     * Value of [enumerated_column] field.
     */
    protected int|null $enumerated_column = null;

    /**
     * Deserialized value of [enumerated_column] field.
     *
     * @var array<string>|null
     */
    protected array|null $enumerated_column_converted = null;

    /**
     * Get the [enumerated_column] column value.
     *
     * @throws \Propel\Runtime\Exception\PropelException
     *
     * @return array|null
     */
    public function getEnumeratedColumn()
    {
        if ($this->enumerated_column_converted === null) {
            $this->enumerated_column_converted = [];
        }
        if (!$this->enumerated_column_converted && $this->enumerated_column !== null) {
            $valueSet = TableTableMap::getValueSet(TableTableMap::COL_ENUMERATED_COLUMN);
            try {
                $this->enumerated_column_converted = SetColumnConverter::convertBitmaskToArray($this->enumerated_column, $valueSet);
            } catch (SetColumnConverterException $e) {
                throw new PropelException('Unknown stored set key: ' . $e->getValue(), $e->getCode(), $e);
            }
        }

        return $this->enumerated_column_converted;
    }

    /**
     * Set the value of [enumerated_column] column.
     *
     * @param array|null $v new value
     *
     * @throws \Propel\Runtime\Exception\PropelException
     *
     * @return $this
     */
    public function setEnumeratedColumn($v)
    {
        if ($this->enumerated_column_converted === null || count(array_diff($this->enumerated_column_converted, $v)) > 0 || count(array_diff($v, $this->enumerated_column_converted)) > 0) {
            $valueSet = TableTableMap::getValueSet(TableTableMap::COL_ENUMERATED_COLUMN);
            try {
                $v = SetColumnConverter::convertToBitmask($v, $valueSet);
            } catch (SetColumnConverterException $e) {
                throw new PropelException(sprintf('Value "%s" is not accepted in this set column', $e->getValue()), $e->getCode(), $e);
            }
            if ($this->enumerated_column !== $v) {
                $this->enumerated_column_converted = null;
                $this->enumerated_column = $v;
                $this->modifiedColumns[TableTableMap::COL_ENUMERATED_COLUMN] = true;
            }
        }

        return $this;
    }
