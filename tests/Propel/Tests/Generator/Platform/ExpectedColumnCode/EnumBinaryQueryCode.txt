
    /**
     * Filter the query on the enumerated_column column
     *
     * @param mixed $enumeratedColumn The value to use as filter
     * @param string|null $comparison Operator to use for the column comparison, defaults to Criteria::EQUAL
     *
     * @throws \Propel\Runtime\Exception\PropelException
     *
     * @return $this
     */
    public function filterByEnumeratedColumn($enumeratedColumn = null, ?string $comparison = null)
    {
        $resolvedColumn = $this->resolveLocalColumnByName('enumerated_column');
        $valueSet = TableTableMap::getValueSet(TableTableMap::COL_ENUMERATED_COLUMN);
        if (is_scalar($enumeratedColumn)) {
            if (!in_array($enumeratedColumn, $valueSet)) {
                throw new PropelException(sprintf('Value "%s" is not accepted in this enumerated column', $enumeratedColumn));
            }
            $enumeratedColumn = array_search($enumeratedColumn, $valueSet);
        } elseif (is_array($enumeratedColumn)) {
            $convertedValues = [];
            foreach ($enumeratedColumn as $value) {
                if (!in_array($value, $valueSet)) {
                    throw new PropelException(sprintf('Value "%s" is not accepted in this enumerated column', $value));
                }
                $convertedValues[] = array_search($value, $valueSet);
            }
            $enumeratedColumn = $convertedValues;
            if ($comparison === null) {
                $comparison = Criteria::IN;
            }
        }
        $this->addUsingOperator($resolvedColumn, $enumeratedColumn, $comparison);

        return $this;
    }
