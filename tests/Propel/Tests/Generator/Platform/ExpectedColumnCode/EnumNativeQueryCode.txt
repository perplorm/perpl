
    /**
     * Filter the query on the enumerated_column column
     *
     * Example usage:
     * <code>
     * $query->filterByEnumeratedColumn('fooValue'); // WHERE enumerated_column = 'fooValue'
     * $query->filterByEnumeratedColumn('%fooValue%', Criteria::LIKE); // WHERE enumerated_column LIKE '%fooValue%'
     * $query->filterByEnumeratedColumn(['foo', 'bar']); // WHERE enumerated_column IN ('foo', 'bar')
     * </code>
     *
     * @param array<string>|string|null $enumeratedColumn The value to use as filter.
     * @param string|null $comparison Operator to use for the column comparison, defaults to Criteria::EQUAL
     *
     * @return $this
     */
    public function filterByEnumeratedColumn($enumeratedColumn = null, ?string $comparison = null)
    {
        $resolvedColumn = $this->resolveLocalColumnByName('enumerated_column');
        if ($comparison === null && is_array($enumeratedColumn)) {
            $comparison = Criteria::IN;
        }
        $this->addUsingOperator($resolvedColumn, $enumeratedColumn, $comparison);

        return $this;
    }
