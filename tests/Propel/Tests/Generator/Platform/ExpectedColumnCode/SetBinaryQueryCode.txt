
    /**
     * Filter the query on the enumerated_column column
     *
     * @param mixed $enumeratedColumn The value to use as filter
     * @param string|null $comparison Operator to use for the column comparison, defaults to Criteria::EQUAL
     *
     * @throws \Propel\Runtime\Exception\PropelException
     *
     * @return $this
     */
    public function filterByEnumeratedColumn($enumeratedColumn = null, ?string $comparison = null)
    {
        $resolvedColumn = $this->resolveLocalColumnByName('enumerated_column');
        $valueSet = TableTableMap::getValueSet(TableTableMap::COL_ENUMERATED_COLUMN);
        try {
            $enumeratedColumn = SetColumnConverter::convertToBitmask($enumeratedColumn, $valueSet);
        } catch (SetColumnConverterException $e) {
            throw new PropelException(sprintf('Value "%s" is not accepted in this set column', $e->getValue()), $e->getCode(), $e);
        }
        if ($comparison === null || $comparison == Criteria::CONTAINS_ALL) {
            if ($enumeratedColumn === 0) {
                return $this;
            }
            $comparison = Criteria::BINARY_ALL;
        } elseif ($comparison == Criteria::CONTAINS_SOME || $comparison == Criteria::IN) {
            if ($enumeratedColumn === 0) {
                return $this;
            }
            $comparison = Criteria::BINARY_AND;
        } elseif ($comparison == Criteria::CONTAINS_NONE) {
            if ($enumeratedColumn !== 0) {
                $this->addAnd($resolvedColumn, $enumeratedColumn, Criteria::BINARY_NONE);
            }
            $this->addOr($resolvedColumn, null, Criteria::ISNULL);

            return $this;
        }
        $this->addUsingOperator($resolvedColumn, $enumeratedColumn, $comparison);

        return $this;
    }
