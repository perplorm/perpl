
    /**
     * Filter the query on the uuid_column column
     *
     * @param mixed $uuidColumn The value to use as filter
     * @param string|null $comparison Operator to use for the column comparison, defaults to Criteria::EQUAL
     *
     * @return $this
     */
    public function filterByUuidColumn($uuidColumn = null, ?string $comparison = null)
    {
        $resolvedColumn = $this->resolveLocalColumnByName('uuid_column');
        $uuidColumn = UuidConverter::uuidToBinRecursive($uuidColumn, true);
        $this->addUsingOperator($resolvedColumn, $uuidColumn, $comparison);

        return $this;
    }
