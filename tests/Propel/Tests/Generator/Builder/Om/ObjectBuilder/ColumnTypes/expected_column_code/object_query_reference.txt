
    /**
     * Filter the query on the object_column column
     *
     * @param mixed $objectColumn The value to use as filter
     * @param string|null $comparison Operator to use for the column comparison, defaults to Criteria::EQUAL
     *
     * @return $this
     */
    public function filterByObjectColumn($objectColumn = null, ?string $comparison = null)
    {
        $resolvedColumn = $this->resolveLocalColumnByName('object_column');
        if (is_object($objectColumn)) {
            $objectColumn = serialize($objectColumn);
        }
        $this->addUsingOperator($resolvedColumn, $objectColumn, $comparison);

        return $this;
    }
