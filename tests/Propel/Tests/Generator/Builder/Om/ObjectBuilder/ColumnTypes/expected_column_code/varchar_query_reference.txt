
    /**
     * Filter the query on the var_char_column column
     *
     * Example usage:
     * <code>
     * $query->filterByVarCharColumn('fooValue'); // WHERE var_char_column = 'fooValue'
     * $query->filterByVarCharColumn('%fooValue%', Criteria::LIKE); // WHERE var_char_column LIKE '%fooValue%'
     * $query->filterByVarCharColumn(['foo', 'bar']); // WHERE var_char_column IN ('foo', 'bar')
     * </code>
     *
     * @param array<string>|string|null $varCharColumn The value to use as filter.
     * @param string|null $comparison Operator to use for the column comparison, defaults to Criteria::EQUAL
     *
     * @return $this
     */
    public function filterByVarCharColumn($varCharColumn = null, ?string $comparison = null)
    {
        $resolvedColumn = $this->resolveLocalColumnByName('var_char_column');
        if ($comparison === null && is_array($varCharColumn)) {
            $comparison = Criteria::IN;
        }
        $this->addUsingOperator($resolvedColumn, $varCharColumn, $comparison);

        return $this;
    }
