
    /**
     * Filter the query on the json_column column
     *
     * Example usage:
     * <code>
     * $query->filterByJsonColumn('fooValue'); // WHERE json_column = 'fooValue'
     * $query->filterByJsonColumn('%fooValue%', Criteria::LIKE); // WHERE json_column LIKE '%fooValue%'
     * $query->filterByJsonColumn(['foo', 'bar']); // WHERE json_column IN ('foo', 'bar')
     * </code>
     *
     * @param array<string>|string|null $jsonColumn The value to use as filter.
     * @param string|null $comparison Operator to use for the column comparison, defaults to Criteria::EQUAL
     *
     * @return $this
     */
    public function filterByJsonColumn($jsonColumn = null, ?string $comparison = null)
    {
        $resolvedColumn = $this->resolveLocalColumnByName('json_column');
        if ($comparison === null && is_array($jsonColumn)) {
            $comparison = Criteria::IN;
        }
        $this->addUsingOperator($resolvedColumn, $jsonColumn, $comparison);

        return $this;
    }
