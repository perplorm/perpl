
    /**
     * Filter the query on the blob_column column
     *
     * @param mixed $blobColumn The value to use as filter
     * @param string|null $comparison Operator to use for the column comparison, defaults to Criteria::EQUAL
     *
     * @return $this
     */
    public function filterByBlobColumn($blobColumn = null, ?string $comparison = null)
    {
        $resolvedColumn = $this->resolveLocalColumnByName('blob_column');
        $this->addUsingOperator($resolvedColumn, $blobColumn, $comparison);

        return $this;
    }
