
    /**
     * Filter the query on the integer_column column
     *
     * Example usage:
     * <code>
     * $query->filterByIntegerColumn(1234); // WHERE integer_column = 1234
     * $query->filterByIntegerColumn(array(12, 34)); // WHERE integer_column IN (12, 34)
     * $query->filterByIntegerColumn(array('min' => 12)); // WHERE integer_column > 12
     * </code>
     *
     * @param mixed $integerColumn The value to use as filter.
     *              Use scalar values for equality.
     *              Use array values for in_array() equivalent.
     *              Use associative array('min' => $minValue, 'max' => $maxValue) for intervals.
     * @param string|null $comparison Operator to use for the column comparison, defaults to Criteria::EQUAL
     *
     * @return $this
     */
    public function filterByIntegerColumn($integerColumn = null, ?string $comparison = null)
    {
        $resolvedColumn = $this->resolveLocalColumnByName('integer_column');
        if (is_array($integerColumn)) {
            $useMinMax = false;
            if (isset($integerColumn['min'])) {
                $this->addUsingOperator($resolvedColumn, $integerColumn['min'], Criteria::GREATER_EQUAL);
                $useMinMax = true;
            }
            if (isset($integerColumn['max'])) {
                $this->addUsingOperator($resolvedColumn, $integerColumn['max'], Criteria::LESS_EQUAL);
                $useMinMax = true;
            }
            if ($useMinMax) {
                return $this;
            }
            if ($comparison === null) {
                $comparison = Criteria::IN;
            }
        }
        $this->addUsingOperator($resolvedColumn, $integerColumn, $comparison);

        return $this;
    }
