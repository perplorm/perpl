    /**

     * to the current object.

     * @param \Propel\Runtime\Collection\Collection<\Team> $leTeams

    public function setLeTeams(Collection $leTeams, ?ConnectionInterface $con = null): static
    {
        $leTeamsToDelete = $this->getLeTeams(null, $con)->diff($leTeams);

        $this->leTeamsScheduledForDeletion = $leTeamsToDelete;

        foreach ($leTeamsToDelete as $leTeamRemoved) {

        $this->collLeTeamsPartial = false;
        $this->collLeTeams = $leTeams instanceof TeamCollection
            ? $leTeams : new TeamCollection($leTeams->getData());

        return $this;
    }

        $leTeam->setUser($this);
    }

    /**
     * @param \Team $leTeam The ChildTeam object to remove.
     *
     * @return static
     */
    public function removeLeTeam(ChildTeam $leTeam): static
    {
        if ($this->getLeTeams()->contains($leTeam)) {
            $pos = $this->collLeTeams->search($leTeam);
            $this->collLeTeams->remove($pos);
            if ($this->leTeamsScheduledForDeletion === null) {
                $this->leTeamsScheduledForDeletion = clone $this->collLeTeams;
                $this->leTeamsScheduledForDeletion->clear();
            }
            $this->leTeamsScheduledForDeletion->append($leTeam);
            $leTeam->setUser(null);
        }

        return $this;
    }
